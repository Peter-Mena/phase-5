task 2: Write scripts/sensor_script.py to simulate data logging (timestamps + random values).
nano sensor_script.py 
#
    import time
    import random
    import os

    sensor_type = os.getenv("temp_sensor", "unknown")

    log_file_path = os.path.expanduser("~/iot_logger/logs/sensor.log")
    data_file_path = os.path.expanduser("~/iot_logger/data/readings.txt")

    with open(log_file_path, "a") as log_file:
        for i in range(10):
            timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
            value = round(random.uniform(20.0, 30.0), 2)

            log_entry = f"{timestamp} - [{sensor_type}] Sensor reading: {value}Â°C\n"

            log_file.write(log_entry)
            log_file.flush() 

            with open(data_file_path, "a") as data_file:
            data_file.write(f"{timestamp},{value}\n")

            time.sleep(2)
#
cd
